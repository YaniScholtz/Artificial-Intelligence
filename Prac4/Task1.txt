prob_R = Count_R / Total_Count if Total_Count > 0 else 0
prob_P = Count_P / Total_Count if Total_Count > 0 else 0


# Calculate conditional probabilities
def calculate_conditional_probabilities(
    Yt, prev_values, total_counts
):  # conditional probs
    conditional_probabilities = {
        "R": {"R": 1, "P": 1, "S": 1},  # Start with 1 to apply Laplace smoothing
        "P": {"R": 1, "P": 1, "S": 1},
        "S": {"R": 1, "P": 1, "S": 1},
    }

    for i in range(len(Yt)):
        if Yt[i] in conditional_probabilities:
            if prev_values[i] in conditional_probabilities[Yt[i]]:
                conditional_probabilities[Yt[i]][prev_values[i]] += 1

    for key in conditional_probabilities:
        total_with_smoothing = total_counts[key] + 3  # Adding 3 for Laplace smoothing
        for subkey in conditional_probabilities[key]:
            conditional_probabilities[key][subkey] /= total_with_smoothing

    return conditional_probabilities


conditional_probabilitiesYt_1 = calculate_conditional_probabilities(
    Yt, Yt_1, {"R": Count_R, "P": Count_P, "S": Count_S}
)

    Yt, Xt_1, {"R": Count_R, "P": Count_P, "S": Count_S}

    history = []

else:
    if matches < 2:  # Less than two moves
        history.append(prev)
        history.append(input)
        prev = np.random.choice(["R", "P", "S"])
        matches += 1
    else:
        hold = prev
        x2, y2, x1, y1 = history[-4], history[-3], history[-2], history[-1]

        p_R = (
            prob_R
            * conditional_probabilitiesYt_2["R"][y2]  # working out P(R|sequence)
            * conditional_probabilitiesXt_2["R"][x2]
            * conditional_probabilitiesYt_1["R"][y1]
            * conditional_probabilitiesXt_1["R"][x1]
        )

        p_P = (
            prob_P
            * conditional_probabilitiesYt_2["P"][y2]
            * conditional_probabilitiesXt_2["P"][x2]
            * conditional_probabilitiesYt_1["P"][y1]
            * conditional_probabilitiesXt_1["P"][x1]
        )

        p_S = (
            prob_S
            * conditional_probabilitiesYt_2["S"][y2]
            * conditional_probabilitiesXt_2["S"][x2]
            * conditional_probabilitiesYt_1["S"][y1]
            * conditional_probabilitiesXt_1["S"][x1]
        )

        probabilities = {"R": p_R, "P": p_P, "S": p_S}
        max_prob = max(probabilities.values())
        max_moves = [move for move, prob in probabilities.items() if prob == max_prob]
        next_move = random.choice(
            max_moves
        )  # if there are two probs that are the same then choose random and next_move is predicted

        if next_move == "R":  # winning outputs
            prev = "P"
        elif next_move == "P":
            prev = "S"
        elif next_move == "S":
            prev = "R"

        history.append(hold)
        history.append(input)
        history = history[2:]
